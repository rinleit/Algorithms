import math 
 
 
class BucketSort: 
 
    def hashing(self, A): 
        m = A[0] 
        for i in range(1, len(A)): 
            if ( m < A[i] ): 
                m = A[i] 
        result = [m, int(math.sqrt(len(A)))] 
        return result 
 
    def re_hashing(self, i, code): 
        return int(i / code[0] * ( code[1] - 1 )) 
 
    def bucketsort(self, A): 
        # get hash codes 
        code = self.hashing(A) 
        buckets = [list() for _ in range(code[1])]
